Integration Guide

To integrate with a frontend application:

  1. Redirect users to: /api/auth/google/login/?redirect_uri=YOUR_CALLBACK_URL
  2. Handle the callback with authorization code
  3. Exchange the code for a token by sending a POST request to /api/auth/google

Example Frontend Code:

// Redirect to Google login
function loginWithGoogle() {
    window.location.href = 
        "http://localhost:8000/api/auth/google/login/?redirect_uri=http://localhost:3000/auth";
}

// Handle the callback
async function handleCallback(code) {
    const response = await fetch('http://localhost:8000/api/auth/google', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            code: code,
            redirect_uri: 'http://localhost:3000/auth'
        })
    });
    
    const data = await response.json();
    localStorage.setItem('token', data.access_token);
}
